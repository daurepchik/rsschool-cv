function partitionOn(pred, items) {
    let arr = [];
    let counter = 0;
    for (let i = 0; i < items.length; i++) {
      if (!pred(items[i])) {
        arr.push(items[i]);
        counter++;
      }
    }
    for (let i = 0; i < items.length; i++) {
      if (pred(items[i])) {
        arr.push(items[i]);
      }
    }
    for (let i in items) {
      items[i] = arr[i];
    }
    return counter;
}
